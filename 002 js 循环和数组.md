##冒泡
冒泡满分玩法:(假设成立法,定义flag假设为true)

    <script>
        var arr = [12, 34, 22, 45, 11, 6, 87, 97, 55, 32, 25, 17, 4, 65];
        var flag = true;
        for (var i = 0; i < arr.length - 1; i++) {
            for (var j = 0; j < arr.length - 1 - i; j++) {
                if (arr[j] > arr[j + 1]) {
                    var temp = arr[j];
                    arr[j]=arr[j+1];
                    arr[j+1]=temp;
                    flag=false;
                }
            }
            if(flag) {
                break;
            }
        }
        console.log(arr);
    </script>
